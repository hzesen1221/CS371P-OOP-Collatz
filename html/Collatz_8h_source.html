00033         //Case 2: arg <= 1,000,000, in cache, simply use cache.

00035         //Case 3: arg <= 1,000,000, NOT in cache.

00037                 if (arg & 1) cache[arg] = 2 + collatz_compute_cyc(arg + (arg >> 1) + 1); //if arg is odd.
